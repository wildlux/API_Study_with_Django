üìö Esercizio Pratico: GET Request
+

Obiettivo: Fare una richiesta GET a un'API pubblica e visualizzare i dati.

import requests

# Definizione dell'URL
url = "https://jsonplaceholder.typicode.com/users"

# Esecuzione della richiesta GET
response = requests.get(url)

# Verifica dello status code
if response.status_code == 200:
    # Stampa della risposta
    print(response.json())
else:
    # Stampa dell'errore
    print("Errore nella richiesta:", response.status_code)
